78 template <typename TSeqs, typename TIds, typename TBarcodes>
79 bool check(TSeqs& seqs, TSeqs& seqsRev, TIds& ids, TBarcodes& barcodes) //returns false on errors with the barcodes, true otherwise
80 {
81  unsigned len = length(barcodes[0]);
82  for(unsigned i = 1; i < length(barcodes); ++i)
83  {
84  if(len != length(barcodes[i]))
85  {
86  std::cerr << "ERROR: Barcodes differ in length. All barcodes must be of equal length.\n";
87  return false; //In this case the programm must be terminated, otherwise erros might occur later or wrong results are likely to be produced
88  }
89  }
90  unsigned limit = length(seqs);
91  unsigned i = 0;
92  while(i < limit)
93  {
94  if(length(seqs[i]) <= len)
95  {
96  std::cout << "WARNING: Sequence " << ids[i] << " has length <= barcode length. Sequence will be excluded.\n";
97  removeValueById(seqs, i);
98  removeValueById(seqsRev, i);
99  removeValueById(ids, i);
100  --limit;
101  }
102  else ++i;
103  }
104  return true;
105 }

331 template <typename TMatches, typename TBarcodes> //must get the StringSet of sequences (NOT Prefices!), the vector containing the matches and the StringSet of barcodes
332 std::vector<std::vector<int> > group(const TMatches& matches, const TBarcodes& barcodes)
333 {
334  std::vector<std::vector<int> > sortedSequences; //initialises the result vector
335  resize(sortedSequences, length(barcodes)+1);
336  for(unsigned i = 0; i < length(matches); ++i)
337  {
338  appendValue(sortedSequences[matches[i]+1], i); //adds intex of sequence to respective group (offset by 1 is necessary since group 0 is reserved for unidentified sequences).
339  }
340  resizeGroups(sortedSequences); //resizes the groups to take minimal space.
341  return sortedSequences;
342 }
