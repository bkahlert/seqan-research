170 template <typename TSeq>
171 unsigned _trimRead(const TSeq& seq, unsigned const cutoff, BWA const &)
172 {
173  int max_arg = length(seq), sum = 0, max = 0;
174  for (int i=length(seq)-1; i >= 0; --i)
175  {
176  sum += cutoff - getQuality(seq, i);
177  if (sum < 0)
178  break;
179  if (sum > max)
180  {
181  max = sum;
182  max_arg = i;
183  }
184  }
185 
186  return max_arg;
187 }

196 // Trim by shifting a window over the sequence and cut where avg. qual. in window turns bad the first time.
197 template <typename TSeq>
198 unsigned _trimRead(const TSeq& seq, unsigned const _cutoff, Mean const & spec)
199 {
200  unsigned window = spec.window;
201  unsigned avg = 0, i = 0;
202  // Work with absolute cutoff in window to avoid divisions.
203  unsigned cutoff = _cutoff*window;
204 
205  // Calculate average quality of initial window.
206  for (i=0; i < window; ++i)
207  avg += getQuality(seq, i);
208 
209  // Shift window over read and keep mean quality, update in constant time.
210  for (i=0; i < length(seq) && avg >= cutoff; ++i)
211  {
212  // Take care only not to go over the end of the sequence. Shorten window near the end.
213  avg -= getQuality(seq, i);
214  avg += i + window < length(seq) ? getQuality(seq, i + window) : 0;
215  }
216 
217  // i now holds the start of the first window that turned bad.
218  return i;
219 }

244 template <typename TSeq, typename TSpec>
245 unsigned trimRead(TSeq& seq, unsigned const cutoff, TSpec const & spec)
246 {
247  unsigned ret, cut_pos;
248  cut_pos = _trimRead(seq, cutoff, spec);
249 
250  ret = length(seq) - cut_pos;
251  erase(seq, cut_pos, length(seq));
252 
253  return ret;

266 template <typename TSet, typename TSpec>
267 unsigned _trimReads(TSet & seqSet, unsigned const cutoff, TSpec const & spec)
268 {
269  typedef typename seqan::Value<TSet>::Type TSeq;
270 
271  int trimmedReads = 0;
272  int len = length(seqSet);
273 
274  #pragma omp parallel for reduction(+:trimmedReads)
275  for (int i=0; i < len; ++i)
276  {
277  TSeq& read = value(seqSet, i);
278  unsigned trimmed = trimRead(read, cutoff, spec);
279  if (trimmed > 0) ++trimmedReads;
280  }
281 
282  return trimmedReads;
283 }

338 template <typename TId, typename TSeq>
339 unsigned dropReads(seqan::StringSet<TId> & idSet1, seqan::StringSet<TSeq> & seqSet1,
340  seqan::StringSet<TId> & idSet2, seqan::StringSet<TSeq> & seqSet2, unsigned const min_length, QualityTrimmingStats& stats)
341 {
342  // Iterate over the set and drop filter out those reads that are
343  // too short. If only one read of the pair is too short, mark it
344  // with a single N. If both reads are too short, remove them.
345  // TODO (Idea for the future generation): Write out orphaned reads to extra file.
346  int i=length(seqSet1) - 1, j=length(seqSet2) - 1;
347  while (i >= 0 && j >= 0)
348  {
349  TSeq& read1 = value(seqSet1, i), read2 = value(seqSet2, i);
350  bool drop1 = length(read1) < min_length;
351  bool drop2 = length(read2) < min_length;
352 
353  if (drop1 && drop2)
354  {
355  ++stats.dropped_1; ++stats.dropped_2;
356  removeValues(idSet1, seqSet1, i);
357  removeValues(idSet2, seqSet2, j);
358  }
359  else if (drop1)
360  {
361  ++stats.dropped_1;
362  seqan::moveValue(seqSet1, i, TSeq("N"));
363  }
364  else if (drop2)
365  {
366  ++stats.dropped_2;
367  seqan::moveValue(seqSet2, j, TSeq("N"));
368  }
369 
370  --i; --j;
371  }
372 
373  return 0;
374 }
