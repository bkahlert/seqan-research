161 template <typename TSeq1, typename TSeq2, bool TTop, bool TLeft, bool TRight, bool TBottom>
162 seqan::Pair<unsigned, seqan::Align<TSeq1> > alignPair(TSeq1& seq1, TSeq2& seq2,
163  const seqan::AlignConfig<TTop, TLeft, TRight, TBottom>& config, bool band = false)
164 {
165  typedef seqan::Align<TSeq1> TAlign;
166  TAlign align;
167  seqan::resize(rows(align), 2);
168  seqan::assignSource(row(align, 0), seq1);
169  seqan::assignSource(row(align, 1), seq2);
170 
171  // Overlap alignment. We again don't allow gaps, since they are unlikely in Illumina data.
172  // We mostly use AlignConfigs true,false,true,false and true,true,true,true here.
173  unsigned score;
174 
175  TScore adapterScore(-100);
176  // If we can do a banded alignment, restrict ourselves to the upper half of the matrix.
177  if (band)
178  score = globalAlignment(align, adapterScore, config, -2, length(seq1));
179  else
180  score = globalAlignment(align, adapterScore, config);
181 
182  return seqan::Pair<unsigned, TAlign>(score, align);
183 }

220 template <typename TAlign>
221 unsigned getInsertSize(TAlign& align)
222 {
223  typedef typename seqan::Row<TAlign>::Type TRow;
224  TRow &row1 = seqan::row(align,0);
225  TRow &row2 = seqan::row(align,1);
226 
227  unsigned seq1_length = length(source(row1));
228  unsigned seq2_length = length(source(row2));
229 
230  // Calculate overlap and overhangs.
231  unsigned overlap = seq1_length - countTotalGaps(row2);
232  unsigned seq2l = seqan::countGaps(seqan::begin(row1)); // Overhang of sequence 2 = Gaps at start of row 1.
233  unsigned seq1r = countTotalGaps(row2) - seqan::countGaps(seqan::begin(row2)); // Overhang of sequence 1 = Gaps at end of row 2.
234 
235  // Insert size: Add sequence lengths, subtract common region (overlap)
236  // and subtract overhangs left and right of insert.
237  return seq1_length + seq2_length - overlap - (seq2l + seq1r);
238 }

249 template <typename TSeq>
250 unsigned stripPair(TSeq& seq1, TSeq& seq2)
251 {
252  // When aligning the two sequences, the complementary sequence is reversed and
253  // complemented, so we have an overlap alignment with complementary bases being the same.
254  typedef typename Value<TSeq>::Type TAlphabet;
255  typedef typename STRING_REVERSE_COMPLEMENT<TAlphabet>::Type TReverseComplement;
256  TReverseComplement mod(seq2);
257 
258  typedef seqan::Align<TSeq> TAlign;
259  seqan::Pair<unsigned, TAlign> ret = alignPair(seq1, mod, seqan::AlignConfig<true, true, true, true>());
260  unsigned score = ret.i1;
261  TAlign align = ret.i2;
262 
263  // Use the overlap of the two sequences to determine the end position.
264  unsigned overlap = getOverlap(align);
265  unsigned mismatches = (overlap-score)/2;
266 
267  // We require a certain correct overlap to exclude spurious hits.
268  // (Especially reverse 3'->5' alignments not caused by adapters.)
269  if (overlap <= 5 || mismatches > overlap*0.15)
270  return 0;
271 
272  // Get actual size of the insert (possible to determine from overlap etc.).
273  unsigned insert = getInsertSize(align);
274 
275  // Now cut both sequences to insert size (no cuts happen if they are smaller)
276  if (length(seq1) > insert)
277  seqan::erase(seq1, insert, length(seq1));
278  if (length(seq2) > insert)
279  seqan::erase(seq2, insert, length(seq2));
280 
281  return insert;
282 }

295 template <typename TSeq>
296 unsigned stripPair(TSeq& seq1, TSeq& adapter1, TSeq& seq2, TSeq& adapter2)
297 {
298  typedef typename Value<TSeq>::Type TAlphabet;
299  typedef typename STRING_REVERSE_COMPLEMENT<TAlphabet>::Type TReverseComplement;
300 
301  // Add adapters to the sequences.
302  seqan::insert(seq1, 0, adapter2);
303  seqan::insert(seq2, 0, adapter1);
304 
305  TReverseComplement mod(seq2);
306 
307  typedef seqan::Align<TSeq> TAlign;
308  seqan::Pair<unsigned, TAlign> ret = alignPair(seq1, mod, seqan::AlignConfig<true, false, true, false>(), true);
309  TAlign align = ret.i2;
310  unsigned score = ret.i1;
311  unsigned overlap = getOverlap(align);
312  unsigned mismatches = (overlap-score)/2;
313 
314  // We can still get the proper insert size from the overlap, since we only
315  // get properly overlapping alignments in this alignment mode. We just have
316  // to subtract the adapter additions. One exception is the case where they
317  // don't overlap at all, but in this case getInsertSize returns 0.
318  unsigned insert = getInsertSize(align);
319  if (insert > 0)
320  insert = insert - length(adapter1) - length(adapter2);
321 
322  // Remove adapter sequences again.
323  seqan::erase(seq1, 0, length(adapter2));
324  seqan::erase(seq2, 0, length(adapter1));
325 
326  // Check the quality of the overlap.
327  if (mismatches >= overlap*0.15)
328  return 0;
329 
330  // Cut down to insert size, if necessary.
331  if (length(seq1) > insert)
332  seqan::erase(seq1, insert, length(seq1));
333  if (length(seq2) > insert)
334  seqan::erase(seq2, insert, length(seq2));
335 
336  return insert;
337 }

348 template <typename TSeq>
349 unsigned stripPairBatch(seqan::StringSet<TSeq>& set1, seqan::StringSet<TSeq>& set2, AdapterTrimmingStats& stats)
350 {
351  int t_num = omp_get_max_threads();
352 
353  // Create local counting variables to avoid concurrency problems.
354  unsigned a1count = 0, a2count = 0, overlapSum = 0;
355  std::vector<unsigned> minOverlap;
356  std::vector<unsigned> maxOverlap;
357  minOverlap.resize(t_num, std::numeric_limits<unsigned>::max());
358  maxOverlap.resize(t_num, 0);
359 
360  int len = length(set1);
361  #pragma omp parallel for reduction(+:a1count, a2count, overlapSum)
362  for (int i=0; i < len; ++i)
363  {
364  // The reads processed in this iteration. The lengths will change after stripPair().
365  TSeq& read1 = value(set1, i), read2 = value(set2, i);
366  unsigned len1 = length(read1), len2 = length(read2);
367 
368  stripPair(read1, read2);
369 
370  // Determine how much was cut off. (Length before and after.)
371  unsigned over1 = len1 - length(read1);
372  unsigned over2 = len2 - length(read2);
373 
374  // Count cuts.
375  a1count += (over1 != 0);
376  a2count += (over2 != 0);
377  overlapSum += over1 + over2; // Count for each mate.
378 
379  // Thread saves local min/max seen in the batch it processed.
380  int t_id = omp_get_thread_num();
381  if (over1 > 0 && std::min(over1,over2) < minOverlap[t_id]) minOverlap[t_id] = std::min(over1,over2);
382  if (over2 > 0 && std::max(over1,over2) > maxOverlap[t_id]) maxOverlap[t_id] = std::max(over1,over2);
383  }
384 
385  // Update statistics using information gathered by the threads.
386  stats.a1count += a1count;
387  stats.a2count += a2count;
388  stats.overlapSum += overlapSum;
389 
390  unsigned batch_min = *std::min_element(minOverlap.begin(), minOverlap.end());
391  unsigned batch_max = *std::max_element(maxOverlap.begin(), maxOverlap.end());
392 
393  if (batch_min < stats.minOverlap) stats.minOverlap = batch_min;
394  if (batch_max > stats.maxOverlap) stats.maxOverlap = batch_max;
395 
396  return 0;
397 }

424  int errors = 0; // No errors for overlap up to 5 bases.

454 template <typename TSeq, typename TAdapter, typename TSpec>
455 unsigned stripAdapter(TSeq& seq, TAdapter& adapter, TSpec& spec)
456 {
457  typedef seqan::Align<TSeq> TAlign;
458  seqan::Pair<int, TAlign> ret = alignAdapter(seq, adapter);
459  int overlap = getOverlap(ret.i2);
460  int score = ret.i1;
461  int mismatches = (overlap-score)/2;
462 
463  if (isMatch(overlap, mismatches, spec))
464  {
465  seqan::erase(seq, length(seq) - overlap, length(seq));
466  return overlap;
467  }
468 
469  return 0;
470 }

484 template <typename TSeq, typename TAdapter, typename TSpec>
485 unsigned stripAdapterBatch(seqan::StringSet<TSeq>& set, TAdapter& adapter, TSpec const & spec,
486  AdapterTrimmingStats& stats, bool reverse = false)
487 {
488  int t_num = omp_get_max_threads();
489 
490  // Create local counting variables to avoid concurrency problems.
491  unsigned a_count = 0, overlapSum = 0;
492  std::vector<unsigned> minOverlap;
493  std::vector<unsigned> maxOverlap;
494  minOverlap.resize(t_num, std::numeric_limits<unsigned>::max());
495  maxOverlap.resize(t_num, 0);
496 
497  int len = length(set);
498  #pragma omp parallel for reduction(+:a_count, overlapSum)
499  for (int i=0; i < len; ++i)
500  {
501  unsigned over = stripAdapter(value(set, i), adapter, spec);
502  overlapSum += over;
503  a_count += (over != 0);
504 
505  // Thread saves local min/max seen in the batch it processed.
506  int t_id = omp_get_thread_num();
507  if (over > 0 && over < minOverlap[t_id]) minOverlap[t_id] = over;
508  if (over > 0 && over > maxOverlap[t_id]) maxOverlap[t_id] = over;
509  }
510 
511  if (!reverse)
512  stats.a1count += a_count;
513  else
514  stats.a2count += a_count;
515  stats.overlapSum += overlapSum;
516 
517  unsigned batch_min = *std::min_element(minOverlap.begin(), minOverlap.end());
518  unsigned batch_max = *std::max_element(maxOverlap.begin(), maxOverlap.end());
519 
520  if (batch_min < stats.minOverlap) stats.minOverlap = batch_min;
521  if (batch_max > stats.maxOverlap) stats.maxOverlap = batch_max;
522 
523  return 0;
524 }
